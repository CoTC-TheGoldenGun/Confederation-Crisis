decreased_crown_authority_effect = {
	# Nothing here in Vanilla
}

increased_crown_authority_effect = {
	any_de_jure_vassal = {
		limit = {
			OR = {
				religion = ROOT
				is_vassal_or_below_of = ROOT
			}
		}

		opinion = {
			who = ROOT
			modifier = opinion_increased_authority
			years = 5
		}
	}
}

decreased_tribal_organization_effect = {
	# Nothing here in Vanilla
}

increased_tribal_organization_effect = {
	any_vassal = {
		limit = {
			is_tribal = yes
			dislike_tribal_organization = yes
		}

		opinion = {
			who = ROOT
			modifier = opinion_increased_tribal_organization
			years = 5
		}
	}
}

#scope of this, is expected to be a ruler character
replace_senators_effect = {
	if = {
		limit = {
			NOT = {	has_law = legislative_authority_0 } # voter appointment law, letting liege to appoint whenever he wants
		}

	    any_courtier_or_vassal = {
			if = {
				limit = {
					is_voter = no
				}
				if = {
					limit = {
						can_hold_title = title_senatorduke
					}
					give_minor_title = title_senatorduke
				}
				else_if = {
					limit = {
						can_hold_title = title_senatorking
					}
					give_minor_title = title_senatorking
				}
				else_if = {
					limit = {
						can_hold_title = title_senatoremperor
					}
					give_minor_title = title_senatoremperor
				}
				else_if = {
					limit = {
						can_hold_title = title_senatorbigrealm1
					}
					give_minor_title = title_senatorbigrealm1
				}
				else_if = {
					limit = {
						can_hold_title = title_senatorbigrealm2
					}
					give_minor_title = title_senatorbigrealm2
				}
				else_if = {
					limit = {
						can_hold_title = title_senatorbigrealm3
					}
					give_minor_title = title_senatorbigrealm3
				}
				else_if = {
					limit = {
						can_hold_title = title_senatorbigrealm4
					}
					give_minor_title = title_senatorbigrealm4
				}
				else_if = {
					limit = {
						can_hold_title = title_senatorbigrealm5
					}
					give_minor_title = title_senatorbigrealm5
				}
			}
		}

		if = {
			limit = {
				OR = {
					can_grant_title = title_senatorduke
					can_grant_title = title_senatorking
					can_grant_title = title_senatoremperor
					can_grant_title = title_senatorbigrealm1
					can_grant_title = title_senatorbigrealm2
					can_grant_title = title_senatorbigrealm3
					can_grant_title = title_senatorbigrealm4
					can_grant_title = title_senatorbigrealm5
				}
			}

			if = {
				limit = { has_law = legislative_authority_1 }
				generate_random_highborn_adult_from_realm_effect = yes
			}
			else = {
				random_list = {
					10 = { generate_random_highborn_adult_from_realm_effect = yes }
					10 = { generate_random_lowborn_adult_from_realm_effect = yes }
				}
			}

			character_event = { id = parliaments.001 days = 1 }
		}
	}
}
