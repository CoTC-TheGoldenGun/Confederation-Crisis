namespace = hunting

# MAIN EVENT CHAIN
# Hunting Trip Begins
narrative_event = {
	id = hunting.0
	title = "EVTNAMEHUNTING0"
	desc = "EVTDESCHUNTING0"
	picture = GFX_evt_hunting_party
	border = GFX_event_narrative_frame_war
	is_triggered_only = yes
	
	immediate = {
		# Find the most important event targets
		event_target:hunting_destination = {
			owner = {
				if = {
					limit = { has_regent = yes }
					regent = { save_event_target_as = planetary_governor }
				}
				else = { save_event_target_as = planetary_governor }
			}
		}
		random_artifact = {
			limit = {
				is_artifact_equipped = yes
				has_artifact_flag = raptor
				NOT = { has_artifact_flag = injured }
				NOT = { has_artifact_flag = sick }
			}
			save_event_target_as = hunting_raptor
		}
		random_artifact = {
			limit = {
				is_artifact_equipped = yes
				has_artifact_flag = weapon
				NOT = { has_artifact_flag = melee }
			}
			save_event_target_as = hunting_weapon
		}
		persistent_event_target:hunting_partner = {
			random_artifact = {
				limit = {
					is_artifact_equipped = yes
					has_artifact_flag = raptor
					NOT = { has_artifact_flag = injured }
					NOT = { has_artifact_flag = sick }
				}
				save_event_target_as = hunting_raptor_2
			}
			random_artifact = {
				limit = {
					is_artifact_equipped = yes
					has_artifact_flag = weapon
					NOT = { has_artifact_flag = melee }
				}
				save_event_target_as = hunting_weapon_2
			}
		}
		# Check if you're hunting with a partner
		clr_character_flag = flag_hunting_with_partner
		if = {
			limit = { persistent_event_target:hunting_partner = { always = yes } }
			set_character_flag = flag_hunting_with_partner
		}
	}
	
	# Go for a standard hunt
	option = {
		name = "EVTOPTAHUNTING0"
		clr_character_flag = flag_hunting_cooldown
		set_character_flag = flag_hunting_cooldown
		clr_character_flag = flag_busy
		clr_character_flag = flag_falconry_hunter
		set_character_flag = flag_on_the_hunt
		clear_event_target = hunting_raptor
		add_trait = on_leave
		if = {
			limit = {
				OR = {
					event_target:planetary_governor = { character = ROOT }
					persistent_event_target:hunting_partner = { character = event_target:planetary_governor }
				}
			}
			narrative_event = { id = hunting.9 days = 5 random = 5 tooltip = "EVTTOOLTIPHUNTING9" }
			hidden_tooltip = { narrative_event = { id = hunting.53 days = 15 random = 30 } }
		}
		else = { event_target:planetary_governor = { narrative_event = { id = hunting.2 tooltip = "EVTTOOLTIPHUNTING2" days = 3 random = 2 } } }
		persistent_event_target:hunting_partner = { narrative_event = { id = hunting.1 tooltip = "EVTTOOLTIPHUNTING1" } }
		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				event_target:hunting_raptor = { always = yes }
				OR = {
					NOT = { persistent_event_target:hunting_partner = { always = yes } }
					event_target:hunting_raptor_2 = { always = yes }
				}
			}
		}
	}
	
	# Go falconing
	option = {
		trigger = {
			event_target:hunting_raptor = { always = yes }
			OR = {
				NOT = { persistent_event_target:hunting_partner = { always = yes } }
				event_target:hunting_raptor_2 = { always = yes }
			}
		}
		name = "EVTOPTBHUNTING0"
		clr_character_flag = flag_hunting_cooldown
		set_character_flag = flag_hunting_cooldown
		clr_character_flag = flag_busy
		set_character_flag = flag_falconry_hunter
		set_character_flag = flag_on_the_hunt
		clear_event_target = hunting_weapon
		add_trait = on_leave
		if = {
			limit = {
				OR = {
					event_target:planetary_governor = { character = ROOT }
					persistent_event_target:hunting_partner = { character = event_target:planetary_governor }
				}
			}
			narrative_event = { id = hunting.9 days = 5 random = 5 tooltip = "EVTTOOLTIPHUNTING9" }
			hidden_tooltip = { narrative_event = { id = hunting.53 days = 15 random = 30 } }
		}
		else = { event_target:planetary_governor = { narrative_event = { id = hunting.2 tooltip = "EVTTOOLTIPHUNTING2" days = 3 random = 2 } } }
		persistent_event_target:hunting_partner = { narrative_event = { id = hunting.1 tooltip = "EVTTOOLTIPHUNTING1" } }
	}
	
	# Nevermind
	option = {
		name = "EVTOPTCHUNTING0"
		clr_character_flag = flag_busy
		ai_chance = { factor = 0 }
	}
}
